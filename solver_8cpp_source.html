88 void find_implications(
89  const std::vector<exprt> &constraints,
90  std::vector<framet> &frames)
91 {
92  const auto frame_map = build_frame_map(frames);
93 
94  for(const auto &c : constraints)
95  {
96  // look for ∀ ς : state . Sxx(ς) ⇒ Syy(...)
97  // and ∀ ς : state . Sxx(ς) ⇒ ...
98  if(c.id() == ID_forall && to_forall_expr(c).where().id() == ID_implies)
99  {
100  auto &implication = to_implies_expr(to_forall_expr(c).where());
101 
102  if(
103  implication.rhs().id() == ID_function_application &&
104  to_function_application_expr(implication.rhs()).function().id() ==
105  ID_symbol)
106  {
107  auto &rhs_symbol = to_symbol_expr(
108  to_function_application_expr(implication.rhs()).function());
109  auto s_it = frame_map.find(rhs_symbol);
110  if(s_it != frame_map.end())
111  {
112  frames[s_it->second.index].implications.emplace_back(
113  implication.lhs(), to_function_application_expr(implication.rhs()));
114  }
115  }
116  }
117  }
118 }

131 std::vector<propertyt> find_properties(
132  const std::vector<exprt> &constraints,
133  const std::vector<framet> &frames)
134 {
135  const auto frame_map = build_frame_map(frames);
136  std::vector<propertyt> properties;
137 
138  for(const auto &c : constraints)
139  {
140  // look for ∀ ς : state . Sxx(ς) ⇒ ...
141  if(c.id() == ID_forall && to_forall_expr(c).where().id() == ID_implies)
142  {
143  auto &implication = to_implies_expr(to_forall_expr(c).where());
144 
145  if(
146  implication.rhs().id() != ID_function_application &&
147  implication.lhs().id() == ID_function_application &&
148  to_function_application_expr(implication.lhs()).function().id() ==
149  ID_symbol)
150  {

153  auto lhs_frame = find_frame(frame_map, lhs_symbol);
154  properties.emplace_back(

156  }
157  }
158  }

203 bool is_subsumed(
204  const std::vector<exprt> &a1,
205  const std::vector<exprt> &a2,
206  const exprt &b,
207  const std::unordered_set<symbol_exprt, irep_hash> &address_taken,
208  bool verbose,
209  const namespacet &ns)
210 {
211  if(b.is_true())
212  return true; // anything subsumes 'true'
213 
214  for(auto &a_conjunct : a1)
215  if(a_conjunct.is_false())
216  return true; // 'false' subsumes anything
217 
218  for(auto &a_conjunct : a1)
219  if(a_conjunct == b)
220  return true; // b is subsumed by a conjunct in a
221 
222  cout_message_handlert message_handler;
223  message_handler.set_verbosity(verbose ? 10 : 1);
224  satcheckt satcheck(message_handler);
225  bv_pointers_widet solver(ns, satcheck, message_handler);
226  axiomst axioms(solver, address_taken, verbose, ns);
227 
228  // check if a => b is valid,
229  // or (!a || b) is valid,
230  // or (a && !b) is unsat
231  for(auto &a_conjunct : a1)
232  axioms << a_conjunct;
233 
234  for(auto &a_conjunct : a2)
235  axioms << a_conjunct;
236 
237  axioms.set_to_false(b);
238 
239  // instantiate our axioms
240  axioms.emit();
241 
242  // now run solver
243  switch(solver())
244  {
245  case decision_proceduret::resultt::D_SATISFIABLE:
246  if(verbose)
247  show_assignment(solver);
248  return false;
249  case decision_proceduret::resultt::D_UNSATISFIABLE:
250  return true;
251  case decision_proceduret::resultt::D_ERROR:
252  throw "error reported by solver";
253  }
254 
255  UNREACHABLE; // to silence a warning
256 }

283 void solver(
284  std::vector<framet> &frames,
285  const std::unordered_set<symbol_exprt, irep_hash> &address_taken,
286  const solver_optionst &solver_options,
287  const namespacet &ns,
288  std::vector<propertyt> &properties,
289  std::size_t property_index)
290 {
291  const auto frame_map = build_frame_map(frames);
292  auto &property = properties[property_index];
293 
294  property.start = std::chrono::steady_clock::now();
295  take_time_resourcet stop_time(property.stop);
296 
297  if(solver_options.verbose)
298  std::cout << "\nDoing " << format(property.condition) << '\n';
299 
300  for(auto &frame : frames)
301  frame.reset();
302 
303  // add properties proven so far as auxiliaries
304  for(std::size_t i = 0; i < property_index; i++)
305  {
306  const auto &p = properties[i];
307  if(p.status == propertyt::PASS)
308  frames[p.frame.index].add_auxiliary(p.condition);
309  }
310 
311  std::vector<workt> queue;
312  std::vector<workt> dropped;
313 
314  auto propagator =
315  [&frames,
316  &frame_map,
317  &queue,
318  &dropped,
319  &address_taken,
320  &solver_options,
321  &ns](
322  const symbol_exprt &symbol, exprt invariant, const workt::patht &path) {
323  auto frame_ref = find_frame(frame_map, symbol);
324  auto &f = frames[frame_ref.index];
325 
326  if(solver_options.verbose)
327  {
328  std::cout << "F: " << format(symbol) << " <- " << format(invariant)
329  << '\n';
330  }
331 
332  // check if already subsumed
333  if(is_subsumed(
334  f.invariants,
335  f.auxiliaries,
336  invariant,
337  address_taken,
338  solver_options.verbose,
339  ns))
340  {
341  if(solver_options.verbose)
342  std::cout << "*** SUBSUMED\n";
343  }
344  else if(count_frame(path, frame_ref) > solver_options.loop_limit)
345  {
346  // loop limit exceeded, drop it
347  if(solver_options.verbose)
348  std::cout << "*** DROPPED\n";
349  dropped.push_back(workt(frame_ref, invariant, path));
350  }
351  else
352  {
353  auto new_path = path;
354  new_path.push_back(frame_ref);
355  queue.emplace_back(f.ref, std::move(invariant), std::move(new_path));
356  }
357  };
358 
359  // we start with I = P
360  queue.emplace_back(
361  property.frame, property.condition, workt::patht{property.frame});
362 
363  while(!queue.empty())
364  {
365  auto work = queue.back();
366  queue.pop_back();
367 
368  frames[work.frame.index].add_invariant(work.invariant);
369 
370  if(solver_options.verbose)
371  {
372  dump(frames, property, true, true);
373  std::cout << '\n';
374  }
375 
376  auto counterexample_found = ::counterexample_found(
377  frames, work, address_taken, solver_options.verbose, ns);
378 
379  if(counterexample_found)
380  {
381  property.status = propertyt::REFUTED;
382  property.trace = counterexample_found.value();
383  return;
384  }
385 
386  propagate(
387  frames, work, address_taken, solver_options.verbose, ns, propagator);
388  }
389 
390  // did we drop anything?
391  if(dropped.empty())
392  property.status = propertyt::PASS;
393  else
394  property.status = propertyt::DROPPED;
395 }

397 solver_resultt solver(
398  const std::vector<exprt> &constraints,
399  const solver_optionst &solver_options,
400  const namespacet &ns)
401 {
402  const auto address_taken = ::address_taken(constraints);
403 
404  if(solver_options.verbose)
405  {
406  for(auto &a : address_taken)
407  std::cout << "address_taken: " << format(a) << '\n';
408  }
409 
410  auto frames = setup_frames(constraints);
411 
412  find_implications(constraints, frames);
413 
414  auto properties = find_properties(constraints, frames);
415 
416  if(properties.empty())
417  {
418  std::cout << "\nThere are no properties to show.\n";
419  return solver_resultt::ALL_PASS;
420  }
421 
422  solver_progresst solver_progress(properties.size(), solver_options.verbose);
423 
424  // solve each property separately, in order of occurence
425  for(std::size_t i = 0; i < properties.size(); i++)
426  {
427  solver_progress(i);
428  solver(frames, address_taken, solver_options, ns, properties, i);
429  }
430 
431  solver_progress.finished();
432 
433  // reporting
434  report_properties(properties);
435 
436  if(solver_options.trace)
437  report_traces(frames, properties, solver_options.verbose, ns);
438 
439  // overall outcome
440  return overall_outcome(properties);
441 }
